import "../styles.css";
import {useEffect, useState} from "react";
import {EuiProvider} from "@elastic/eui";

export default function Nextra({Component, pageProps}) {
  const [isClient, setIsClient] = useState(false);
  useEffect(() => {
    setIsClient(true);
  }, []);
  if (isClient) {
    return (
      <EuiProvider colorMode={"dark"}>
          <Component {...pageProps} />
      </EuiProvider>
    );
  }
  return null;
}
